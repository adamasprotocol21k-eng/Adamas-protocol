function openGame(id) { document.getElementById(id).style.display = 'flex'; if(id === 'scratchBox') initScratch(); if(id === 'minesBox') initMines(); } function closeModal(id) { document.getElementById(id).style.display = 'none'; } // --- MINES LOGIC (Betting) --- let mines = []; let gameActive = false; function initMines() { const board = document.getElementById('minesBoard'); board.innerHTML = ''; for(let i=0; i<25; i++) { const t = document.createElement('div'); t.className = 'tile'; t.onclick = () => revealTile(i, t); board.appendChild(t); } } function startMines() { if(window.userStats.balance < 100) return showNotification("Min Bet 100 ABP Required!"); window.userStats.balance -= 100; saveToFirebase(); mines = Array.from({length: 3}, () => Math.floor(Math.random() * 25)); gameActive = true; initMines(); showNotification("Game Started! 100 ABP Bet."); } function revealTile(idx, el) { if(!gameActive || el.classList.contains('revealed')) return; el.classList.add('revealed'); if(mines.includes(idx)) { el.innerHTML = '💣'; el.style.background = 'red'; gameActive = false; showNotification("BOOM! You lost the bet."); setTimeout(() => closeModal('minesBox'), 1500); } else { el.innerHTML = '💎'; window.userStats.balance += 150; // Win logic saveToFirebase(); } } // --- SCRATCH LOGIC --- let scratchVal = 0; function initScratch() { const canvas = document.getElementById('scratchCanvas'); const ctx = canvas.getContext('2d'); ctx.globalCompositeOperation = 'source-over'; ctx.fillStyle = '#888'; ctx.fillRect(0,0,300,150); scratchVal = Math.floor(Math.random()*500)+50; document.getElementById('scratchReward').innerText = scratchVal + " ABP"; canvas.ontouchmove = (e) => { const rect = canvas.getBoundingClientRect(); const x = e.touches[0].clientX - rect.left; const y = e.touches[0].clientY - rect.top; ctx.globalCompositeOperation = 'destination-out'; ctx.beginPath(); ctx.arc(x,y,25,0,Math.PI*2); ctx.fill(); } } function collectScratchReward() { if(scratchVal > 0) { window.userStats.balance += scratchVal; saveToFirebase(); showNotification(`Added ${scratchVal} ABP!`); scratchVal = 0; } closeModal('scratchBox'); }
